def initialize(options)
  opts = options.dup

  @tls = true

  url = opts.delete(:url) || ENV["MALONE_URL"]

  if url
    uri = URI(url)

    opts[:host]     ||= uri.host
    opts[:port]     ||= uri.port.to_i
    opts[:user]     ||= unescaped(uri.user)
    opts[:password] ||= unescaped(uri.password)
  end

  opts.each do |key, val|
    send(:"#{key}=", val)
  end
end

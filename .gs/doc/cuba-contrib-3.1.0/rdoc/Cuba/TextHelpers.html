def titlecase(str)
  res = str.to_s.dup
  res.tr!("_", " ")
  res.gsub!(%r(^|\s)([a-z])/) { |char| char.upcase }
  res
end

def truncate(str, length, ellipses = "...")
  return str if !str || str.length <= length

  sprintf("%.#{length}s#{ellipses}", str)
end

def underscore(str)
  res = str.to_s.dup
  res.gsub!(%r([A-Z]+)([A-Z][a-z])/,'\1_\2')
  res.gsub!(%r([a-z\d])([A-Z])/,'\1_\2')
  res.downcase!
  res
end
